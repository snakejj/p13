13 def _parse_link(self, request, raw_link): 

14 # Examples: 

15 # - http://youtu.be/SA2iWivDJiE 

16 # - http://www.youtube.com/watch?v=_oPAwA_Udwc&feature=feedu 

17 # - http://www.youtube.com/embed/SA2iWivDJiE 

18 # - http://www.youtube.com/v/SA2iWivDJiE?version=3&amp;hl=en_US 

19 query = urlparse(raw_link) 

20 if query.hostname == 'youtu.be': 

21 clean = query.path[1:] 

22 if len(clean) == 11: 

23 return clean 

24 if query.hostname in {'www.youtube.com', 'youtube.com'}: 

25 if query.path == '/watch': 

26 clean = parse_qs(query.query)['v'][0] 

27 if len(clean) == 11: 

28 return clean 

29 if query.path[:7] == '/embed/': 

30 clean = query.path.split('/')[2] 

31 if len(clean) == 11: 

32 return clean 

33 if query.path[:3] == '/v/': 

34 clean = query.path.split('/')[2] 

35 if len(clean) == 11: 

36 return clean 

37 elif request.user.is_authenticated: 

38 return True 

39 # fail? 

40 return False

256 list_interest_ratings.append(interest_rating) 

257 list_quality_ratings.append(quality_rating) 

258 

259 average_interest_rating = sum(list_interest_ratings) / len(list_interest_ratings) 

260 average_quality_rating = sum(list_quality_ratings) / len(list_quality_ratings) 

261 else: 

262 average_interest_rating = interest_rating
